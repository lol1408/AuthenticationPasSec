//Авторизация
URL : https://domain/login
    Request:
        METHOD: GET,
        HEADERS:
        {
            'Accept' : 'application/json',
            'Content-Type' : 'application/json',
            'login' : '${login}',
            'password' : '${encrypt_password}'
        }
    Response:
        //Если вход прошел успешно, P.S. token - целочисленный
        STATUS: 200
        BODY : ${token}
        //Если аккаунт не активна
        STATUS: 423
        //Если аккаунт не найден
        STATUS: 404
